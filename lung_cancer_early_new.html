2.Clean data

# recode event variable
lungcancer$survival_months=as.numeric(as.character(lungcancer$Survival.months))
lungcancer$event=case_when(lungcancer$SEER.cause.specific.death.classification=="Alive or dead of other cause"
                           |lungcancer$SEER.cause.specific.death.classification=="Dead (missing/unknown COD)"~0,
                           lungcancer$SEER.cause.specific.death.classification=="Dead (attributable to this cancer dx)"~1)

# recode marriage variable
lungcancer$marital_cat=case_when(lungcancer$Marital.status.at.diagnosis=="Divorced"|lungcancer$Marital.status.at.diagnosis=="Separated"|lungcancer$Marital.status.at.diagnosis=="Single (never married)"|lungcancer$Marital.status.at.diagnosis=="Widowed"|lungcancer$Marital.status.at.diagnosis=="Unmarried or Domestic Partner"~1,lungcancer$Marital.status.at.diagnosis=="Married (including common law)"~2)
lungcancer$marital_cat=factor(lungcancer$marital_cat,1:2,c("unmarried","married"))

# recode age variable
lungcancer$age_cat=case_when(lungcancer$Age.recode.with..1.year.olds=="10-14 years"|lungcancer$Age.recode.with..1.year.olds=="20-24 years"|
                               lungcancer$Age.recode.with..1.year.olds=="25-29 years"|lungcancer$Age.recode.with..1.year.olds=="30-34 years"|
                               lungcancer$Age.recode.with..1.year.olds=="35-39 years"|lungcancer$Age.recode.with..1.year.olds=="40-44 years"~1,
                             lungcancer$Age.recode.with..1.year.olds=="45-49 years"|lungcancer$Age.recode.with..1.year.olds=="50-54 years"~2,
                             lungcancer$Age.recode.with..1.year.olds=="55-59 years"|lungcancer$Age.recode.with..1.year.olds=="60-64 years"~3,
                             lungcancer$Age.recode.with..1.year.olds=="65-69 years"|lungcancer$Age.recode.with..1.year.olds=="70-74 years"~4,
                             lungcancer$Age.recode.with..1.year.olds=="75-79 years"|lungcancer$Age.recode.with..1.year.olds=="80-84 years"~5,
                             lungcancer$Age.recode.with..1.year.olds=="85+ years"~6)
lungcancer$age_cat=factor(lungcancer$age_cat,1:6,c("<45 years","45-54 years","55-64 years","65-74 years","75-84 years",">85 years"))

# recode race/ethnicity variable
lungcancer$race_cat=case_when(lungcancer$Race.and.origin.recode..NHW..NHB..NHAIAN..NHAPI..Hispanic.=="Non-Hispanic White"~1,
                              lungcancer$Race.and.origin.recode..NHW..NHB..NHAIAN..NHAPI..Hispanic.=="Non-Hispanic Black"~2,
                              lungcancer$Race.and.origin.recode..NHW..NHB..NHAIAN..NHAPI..Hispanic.=="Non-Hispanic Asian or Pacific Islander"~3,
                              lungcancer$Race.and.origin.recode..NHW..NHB..NHAIAN..NHAPI..Hispanic.=="Non-Hispanic American Indian/Alaska Native"~4,
                              lungcancer$Race.and.origin.recode..NHW..NHB..NHAIAN..NHAPI..Hispanic.=="Hispanic (All Races)"~5)
lungcancer$race_cat=factor(lungcancer$race_cat,1:5,c("White","black","Asian","American Indian","Hispanic"))


# remove all observation with unknown (missing)
final=na.omit(lungcancer)

# create a new dataset containing only patients diagnosed in 2000-01
lung_early=final[which(final$Year.of.diagnosis==2000|final$Year.of.diagnosis==2001),]

# create a new dataset containing only patients diagnosed in 2010-2011
lung_new=final[which(final$Year.of.diagnosis==2011 | final$Year.of.diagnosis==2010),]
